//可以通过编程的方式随意动态的增加字段的校验规则
//Field.addRule() : 增加校验规则
//Field.removeRule() : 移除校验规则

		S.one("#optrule").on("click",function(ev){
			ev.halt();
			
			//移除email校验
			username.removeRule("email");
			
			//增加中文校验
			username.addRule("chinese",['必须为中文']);
			
			//触发校验
			username.isValid();
			
		});
