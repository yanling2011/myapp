KISSY.use("validation",function(S,Validation){

			var form = S.query("form")[0],
			
				check = new Validation(form,{
					style: "under"
				});

			//实例化字段，配置通过参数传入
			var f = new Validation.Field("#username",{
				style:'under',
				chinese:['姓名只能是中文']
			})
	
			//增加到校验中
			check.add(f);


			KISSY.Event.on(form,"submit",function(){
				check.isValid();
				return false;
			});
		
	});

  • 如果不想通过伪属性配置，可以实例化Validation后用add方法增加要校验的字段
